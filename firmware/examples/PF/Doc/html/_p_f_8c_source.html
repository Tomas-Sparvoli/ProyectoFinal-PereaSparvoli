144void cambioEstado(void *param) {
145 while(true) {
146 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
147 uint8_t caracter;
148 UartReadByte(UART_PC, &caracter);
149
150 if (caracter == 'a') {
151 giro = true;
152 vTaskNotifyGiveFromISR(cambiomotor, pdFALSE);
153 } else if (caracter == 'm') {
154 mapa = !mapa;
155 enviar_datos();
156 current_led = 10;
157 current_angle = 110;
158 } else if (caracter == 'b') {
159 color = NEOPIXEL_COLOR_WHITE;
160 printf("Color seleccionado: blanco\n");
161 } else if (caracter == 'r') {
162 color = NEOPIXEL_COLOR_RED;
163 printf("Color seleccionado: rojo\n");
164 } else if (caracter == 'l') {
165 LedControlTask();
166 }
167 }
168}
